OPTION 1: Automatic (Recommended)
  $ cd klippboard
  $ bash install.sh
  $ source ~/.bashrc
  $ klippboard

OPTION 2: Manual
  $ python3 klippboard/src/clipboard_manager.py

OPTION 3: With Virtual Env
  $ python3 -m venv ~/clipboard_env
  $ source ~/clipboard_env/bin/activate
  $ pip install PyQt5 keyboard
  $ python3 klippboard/src/clipboard_manager.py
